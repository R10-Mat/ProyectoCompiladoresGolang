func main() {
    var arr [500]int;
    var i int;
    var j int;
    var temp int;

    for i = 0; i < 500; i++ {
        arr[i] = 500 - i;
    };

    for i = 0; i < 500; i++ {
        for j = 0; j < 499; j++ {
            if arr[j] > arr[j + 1] {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            };
        };
    };
};
