func main() {
    var count int = 0;
    var estado int;
    var i int;

    for i = 0; i < 1000000; i++ {
        estado = i % 4;
        
        switch estado {
            case 0:
                count += 1;
            case 1:
                count += 2;
            case 2:
                count -= 1;
            default:
                count += 3;
        };
        
        if count > 10000 {
            count = 0;
        } else {
            count = count + 1;
        };
    };
};
