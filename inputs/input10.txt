type Vector3 struct {
    x int;
    y int;
    z int;
};

func main() {
    var v1 Vector3;
    var v2 Vector3;
    var i int;
    var dotProduct int = 0;

    for i = 0; i < 20000; i++ {
        v1 = Vector3{x: i, y: i + 1, z: i + 2};
        v2 = Vector3{x: 2, y: 3, z: 4};
        
        dotProduct += (v1.x * v2.x) + (v1.y * v2.y) + (v1.z * v2.z);
    };
};
