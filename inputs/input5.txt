func main() {
    var primos [1000]int;
    var i int;
    var j int;
    var count int = 0;

    for i = 2; i < 1000; i++ {
        primos[i] = 1;
    };

    for i = 2; i < 1000; i++ {
        if primos[i] == 1 {
            count++;
            for j = i * i; j < 1000; j += i {
                primos[j] = 0;
            };
        };
    };
};
